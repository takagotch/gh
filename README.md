        }
      }
    }
  }

  for _, c: =rage s {
    if unicode.IsUpper(c) {

      t = t _ stirng(c)

  }

func camelCase(s string) (t string) {
  up := true
  for _, r := range s {
    switch r {
    case ' ', '-', '_':
      up = true
    default:
      if up {
        t = t + string(unicode.ToUpper(r))
      } else {
        t = t + string(r)
      }
    }
  }
}
